26. Longest Subarray with Sum K
Arrays + HashMap / Two Pointers / Prefix Sum
Approach
  • Prefix sum + first occurrence index.
  • If (ps-k) seen at index j, length = i-j.
  • Store earliest index for each prefix sum.
Complexity: O(n) time, O(n) space.
Java
import java.util.*;

public class LongestSubarraySumK {

    public static int longest(int[] nums, int k) {
        Map<Integer, Integer> first = new HashMap<>();
        first.put(0, -1);

        int ps = 0, best = 0;

        for (int i = 0; i < nums.length; i++) {
            ps += nums[i];
            first.putIfAbsent(ps, i);

            Integer j = first.get(ps - k);
            if (j != null) best = Math.max(best, i - j);
        }
        return best;
    }
}
27. Count Pairs with Given Sum
Arrays + HashMap / Two Pointers / Prefix Sum
Approach
  • Use HashMap counts.
  • For each x, add count(target-x) to answer, then increment count(x).
Complexity: O(n) time, O(n) space.
Java
import java.util.*;

public class CountPairsSum {

    public static long countPairs(int[] nums, int target) {
        Map<Integer, Long> cnt = new HashMap<>();
        long ans = 0;

        for (int x : nums) {
            ans += cnt.getOrDefault(target - x, 0L);
            cnt.put(x, cnt.getOrDefault(x, 0L) + 1);
        }
        return ans;
    }
}
28. Move All Zeros to End
Arrays + HashMap / Two Pointers / Prefix Sum
Approach
  • Two pointers: write index for non-zeros.
  • Write non-zeros forward, then fill the rest with zeros.
Complexity: O(n) time, O(1) space.
Java
public class MoveZeros {

    public static void moveZeroes(int[] nums) {
        int w = 0;

        for (int x : nums) {
            if (x != 0) nums[w++] = x;
        }

        while (w < nums.length) nums[w++] = 0;
    }
}
29. Rotate Array by K Positions
Arrays + HashMap / Two Pointers / Prefix Sum
Approach
  • Normalize k = k % n.
  • Reverse whole array, then reverse [0..k-1], then reverse [k..n-1].
Complexity: O(n) time, O(1) space.
Java
public class RotateArray {

    public static void rotate(int[] a, int k) {
        int n = a.length;
        if (n == 0) return;

        k %= n;

        reverse(a, 0, n - 1);
        reverse(a, 0, k - 1);
        reverse(a, k, n - 1);
    }

    private static void reverse(int[] a, int l, int r) {
        while (l < r) {
            int tmp = a[l];
            a[l] = a[r];
            a[r] = tmp;
            l++;
            r--;
        }
    }
}
30. Find Minimum in Rotated Sorted Array
Arrays + HashMap / Two Pointers / Prefix Sum
Approach
  • Binary search comparing mid with right.
  • If mid > right, min is to the right; else to the left (including mid).
Complexity: O(log n) time, O(1) space.
Java
public class MinRotatedSorted {

    public static int findMin(int[] nums) {
        int l = 0, r = nums.length - 1;

        while (l < r) {
            int m = l + (r - l) / 2;

            if (nums[m] > nums[r]) l = m + 1;
            else r = m;
        }
        return nums[l];
    }
}
