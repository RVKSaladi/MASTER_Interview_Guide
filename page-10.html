46. Evaluate Postfix Expression (RPN)
Stack / Queue
Approach
  • Use stack of ints.
  • When operator seen, pop two operands, apply, push result.
Complexity: O(n) time, O(n) space.
Java
import java.util.*;

public class EvalRPN {

    public static int evalRPN(String[] tokens) {
        Deque<Integer> st = new ArrayDeque<>();

        for (String t : tokens) {
            if (t.equals("+") || t.equals("-") || t.equals("*") || t.equals("/")) {
                int b = st.pop(), a = st.pop();
                int r;

                if (t.equals("+")) r = a + b;
                else if (t.equals("-")) r = a - b;
                else if (t.equals("*")) r = a * b;
                else r = a / b; // trunc toward 0

                st.push(r);
            } else {
                st.push(Integer.parseInt(t));
            }
        }
        return st.pop();
    }
}
47. Stock Span Problem
Stack / Queue
Approach
  • Monotonic decreasing stack of indices (prices).
  • Pop while current price >= stackTop price.
  • Span = i - newTopIndex (or i+1 if empty).
Complexity: O(n) time, O(n) space.
Java
import java.util.*;

public class StockSpan {

    public static int[] span(int[] prices) {
        int n = prices.length;
        int[] res = new int[n];
        Deque<Integer> st = new ArrayDeque<>();

        for (int i = 0; i < n; i++) {
            while (!st.isEmpty() && prices[i] >= prices[st.peek()]) st.pop();
            res[i] = st.isEmpty() ? (i + 1) : (i - st.peek());
            st.push(i);
        }
        return res;
    }
}
48. Daily Temperatures
Stack / Queue
Approach
  • Monotonic decreasing stack of indices (temps).
  • When warmer day arrives, resolve previous indices.
Complexity: O(n) time, O(n) space.
Java
import java.util.*;

public class DailyTemperatures {

    public static int[] dailyTemperatures(int[] t) {
        int n = t.length;
        int[] res = new int[n];
        Deque<Integer> st = new ArrayDeque<>();

        for (int i = 0; i < n; i++) {
            while (!st.isEmpty() && t[i] > t[st.peek()]) {
                int j = st.pop();
                res[j] = i - j;
            }
            st.push(i);
        }
        return res;
    }
}
49. Sliding Window Maximum
Stack / Queue
Approach
  • Use deque of indices with values in decreasing order.
  • Pop back while current >= back value.
  • Front is max; remove front if it leaves window.
Complexity: O(n) time, O(n) space.
Java
import java.util.*;

public class SlidingWindowMax {

    public static int[] maxSlidingWindow(int[] nums, int k) {
        int n = nums.length;
        if (k <= 0 || n == 0) return new int[0];

        int[] res = new int[n - k + 1];
        Deque<Integer> dq = new ArrayDeque<>();

        for (int i = 0; i < n; i++) {
            while (!dq.isEmpty() && dq.peekFirst() <= i - k) dq.removeFirst();
            while (!dq.isEmpty() && nums[i] >= nums[dq.peekLast()]) dq.removeLast();
            dq.addLast(i);

            if (i >= k - 1) res[i - k + 1] = nums[dq.peekFirst()];
        }
        return res;
    }
}
50. Simplify Unix File Path
Stack / Queue
Approach
  • Split by '/'.
  • Use stack: ignore '' and '.', pop for '..', push normal dir names.
  • Join with '/'.
Complexity: O(n) time, O(n) space.
Java
import java.util.*;

public class SimplifyPath {

    public static String simplify(String path) {
        Deque<String> st = new ArrayDeque<>();

        for (String part : path.split("/")) {
            if (part.isEmpty() || part.equals(".")) continue;
            if (part.equals("..")) {
                if (!st.isEmpty()) st.pop();
            } else {
                st.push(part);
            }
        }

        StringBuilder sb = new StringBuilder();
        Iterator<String> it = st.descendingIterator();
        while (it.hasNext()) sb.append('/').append(it.next());

        return sb.length() == 0 ? "/" : sb.toString();
    }
}
