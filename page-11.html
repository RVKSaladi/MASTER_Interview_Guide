51. Inorder Traversal (Iterative)
Trees / BST
Approach
  • Use stack; push left chain.
  • Pop node, visit, then go right.
Complexity: O(n) time, O(h) space.
Java
import java.util.*;

public class InorderTraversal {

    static class TreeNode {
        int val;
        TreeNode left, right;
        TreeNode(int v) { val = v; }
    }

    public static List<Integer> inorder(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        Deque<TreeNode> st = new ArrayDeque<>();
        TreeNode cur = root;

        while (cur != null || !st.isEmpty()) {
            while (cur != null) {
                st.push(cur);
                cur = cur.left;
            }
            cur = st.pop();
            res.add(cur.val);
            cur = cur.right;
        }
        return res;
    }
}
52. Level Order Traversal
Trees / BST
Approach
  • BFS with queue.
  • Process nodes level by level (track size).
Complexity: O(n) time, O(n) space.
Java
import java.util.*;

public class LevelOrder {

    static class TreeNode {
        int val;
        TreeNode left, right;
        TreeNode(int v) { val = v; }
    }

    public static List<List<Integer>> levelOrder(TreeNode root) {
        List<List<Integer>> res = new ArrayList<>();
        if (root == null) return res;

        Deque<TreeNode> q = new ArrayDeque<>();
        q.add(root);

        while (!q.isEmpty()) {
            int sz = q.size();
            List<Integer> level = new ArrayList<>();

            for (int i = 0; i < sz; i++) {
                TreeNode n = q.remove();
                level.add(n.val);
                if (n.left != null) q.add(n.left);
                if (n.right != null) q.add(n.right);
            }
            res.add(level);
        }
        return res;
    }
}
53. Height of Binary Tree
Trees / BST
Approach
  • Recursive: height = 1 + max(left, right).
  • Base case: null -> 0.
Complexity: O(n) time, O(h) stack.
Java
public class TreeHeight {

    static class TreeNode {
        int val;
        TreeNode left, right;
        TreeNode(int v) { val = v; }
    }

    public static int height(TreeNode root) {
        if (root == null) return 0;
        return 1 + Math.max(height(root.left), height(root.right));
    }
}
54. Check if Tree is Balanced
Trees / BST
Approach
  • Compute height bottom-up.
  • If any subtree differs by > 1, return -1 as sentinel.
Complexity: O(n) time, O(h) stack.
Java
public class BalancedTree {

    static class TreeNode {
        int val;
        TreeNode left, right;
        TreeNode(int v) { val = v; }
    }

    public static boolean isBalanced(TreeNode root) {
        return dfs(root) != -1;
    }

    private static int dfs(TreeNode n) {
        if (n == null) return 0;

        int lh = dfs(n.left);
        if (lh == -1) return -1;

        int rh = dfs(n.right);
        if (rh == -1) return -1;

        if (Math.abs(lh - rh) > 1) return -1;
        return 1 + Math.max(lh, rh);
    }
}
55. Diameter of Binary Tree
Trees / BST
Approach
  • DFS returns height.
  • Update global diameter with leftHeight + rightHeight.
Complexity: O(n) time, O(h) stack.
Java
public class DiameterTree {

    static class TreeNode {
        int val;
        TreeNode left, right;
        TreeNode(int v) { val = v; }
    }

    private static int best;

    public static int diameter(TreeNode root) {
        best = 0;
        height(root);
        return best;
    }

    private static int height(TreeNode n) {
        if (n == null) return 0;

        int lh = height(n.left);
        int rh = height(n.right);

        best = Math.max(best, lh + rh);
        return 1 + Math.max(lh, rh);
    }
}
