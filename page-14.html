66. Combination Sum (Reuse Elements Allowed)
Recursion / Backtracking
Approach
  • Sort candidates.
  • Backtrack with current sum; allow reusing same index.
  • Stop when sum == target or sum > target.
Complexity: Exponential backtracking.
Java
import java.util.*;

public class CombinationSum {

    public static List<List<Integer>> combinationSum(int[] cand, int target) {
        Arrays.sort(cand);
        List<List<Integer>> res = new ArrayList<>();
        dfs(cand, 0, target, new ArrayList<>(), res);
        return res;
    }

    private static void dfs(int[] a, int start, int target, List<Integer> cur, List<List<Integer>> res) {
        if (target == 0) {
            res.add(new ArrayList<>(cur));
            return;
        }

        for (int i = start; i < a.length; i++) {
            if (a[i] > target) break;
            cur.add(a[i]);
            dfs(a, i, target - a[i], cur, res);
            cur.remove(cur.size() - 1);
        }
    }
}
67. Rat in a Maze (Find One Path)
Recursion / Backtracking
Approach
  • Backtrack from (0,0) to (n-1,n-1).
  • Move in 4 directions; mark visited.
  • Return when destination reached.
Complexity: Exponential worst-case.
Java
public class RatInMaze {

    private static final int[] dr = {1, -1, 0, 0};
    private static final int[] dc = {0, 0, 1, -1};

    public static boolean solve(int[][] maze, int r, int c, boolean[][] vis) {
        int n = maze.length;

        if (r < 0 || c < 0 || r >= n || c >= n) return false;
        if (maze[r][c] == 0 || vis[r][c]) return false;
        if (r == n - 1 && c == n - 1) return true;

        vis[r][c] = true;

        for (int k = 0; k < 4; k++) {
            if (solve(maze, r + dr[k], c + dc[k], vis)) return true;
        }

        vis[r][c] = false;
        return false;
    }
}
68. Sudoku Solver
Recursion / Backtracking
Approach
  • Backtrack on empty cells.
  • Try digits 1..9 that keep row/col/box valid.
  • Recurse; if fail, reset cell.
Complexity: Backtracking; fast in practice with pruning.
Java
public class SudokuSolver {

    public static void solveSudoku(char[][] board) {
        solve(board);
    }

    private static boolean solve(char[][] b) {
        for (int r = 0; r < 9; r++) {
            for (int c = 0; c < 9; c++) {
                if (b[r][c] == '.') {
                    for (char d = '1'; d <= '9'; d++) {
                        if (ok(b, r, c, d)) {
                            b[r][c] = d;
                            if (solve(b)) return true;
                            b[r][c] = '.';
                        }
                    }
                    return false;
                }
            }
        }
        return true;
    }

    private static boolean ok(char[][] b, int r, int c, char d) {
        for (int i = 0; i < 9; i++) {
            if (b[r][i] == d) return false;
            if (b[i][c] == d) return false;
        }

        int br = (r / 3) * 3, bc = (c / 3) * 3;
        for (int i = br; i < br + 3; i++) {
            for (int j = bc; j < bc + 3; j++) {
                if (b[i][j] == d) return false;
            }
        }
        return true;
    }
}
69. Print Even and Odd Numbers Using Two Threads
Concurrency / Multithreading
Approach
  • Share a lock and current number.
  • Odd thread waits while number is even; even thread waits while number is odd.
  • Use wait/notifyAll.
Complexity: O(n) time.
Java
public class EvenOddPrinter {

    private final Object lock = new Object();
    private int cur = 1;
    private final int max;

    public EvenOddPrinter(int max) {
        this.max = max;
    }

    public void printOdd() throws InterruptedException {
        while (true) {
            synchronized (lock) {
                while (cur <= max && (cur % 2 == 0)) {
                    lock.wait();
                }
                if (cur > max) {
                    lock.notifyAll();
                    return;
                }
                System.out.println("Odd: " + cur++);
                lock.notifyAll();
            }
        }
    }

    public void printEven() throws InterruptedException {
        while (true) {
            synchronized (lock) {
                while (cur <= max && (cur % 2 == 1)) {
                    lock.wait();
                }
                if (cur > max) {
                    lock.notifyAll();
                    return;
                }
                System.out.println("Even: " + cur++);
                lock.notifyAll();
            }
        }
    }
}
70. Producer–Consumer Problem (Bounded Buffer)
Concurrency / Multithreading
Approach
  • Use a BlockingQueue with capacity.
  • Producer puts; consumer takes.
  • BlockingQueue handles waiting safely.
Complexity: O(ops) time.
Java
import java.util.concurrent.*;

public class ProducerConsumer {

    public static void main(String[] args) {
        BlockingQueue<Integer> q = new ArrayBlockingQueue<>(10);

        Runnable producer = () -> {
            try {
                for (int i = 0; i < 100; i++) q.put(i);
            } catch (InterruptedException ignored) {}
        };

        Runnable consumer = () -> {
            try {
                for (int i = 0; i < 100; i++) System.out.println(q.take());
            } catch (InterruptedException ignored) {}
        };

        new Thread(producer).start();
        new Thread(consumer).start();
    }
}
